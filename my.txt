function Timer() {
	var clock=$("#clock").html().split(':');
	if (clock[1]<=0){
		clock[1]='59';
		clock[0]=clock[0]-1;
	} 
	else {
		clock[1]=clock[1]-1;
	}
	$("#clock").html(clock[0]+':'+clock[1]);
    n = setTimeout(arguments.callee, 1000);
}
function plus_(timout) {
	var clock=$("#clock").html()*1;
	if (clock<=0){
		return
	}
	else {
		clock=clock+1;
	
	$("#clock").html(clock);
    n = setTimeout(arguments.callee, 30000);
	$("#stop_task").attr('handler',n);
	
	}
}
